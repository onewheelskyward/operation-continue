def pretty_print_header_text(text, width)
    pre_dashes = 5
    text =  " #{text} "
    right_side_dashes_width = width - text.length - pre_dashes
    puts ['|',
          '-' * pre_dashes,
          text,
          '-' * right_side_dashes_width,
          '|']
          .pack("A1A5A#{text.length}A#{right_side_dashes_width}A1")
end

def pretty_print_header_char(edge_char, char, width)
    puts [edge_char, char * width, edge_char].pack("A1A#{width}A1")
end

def pretty_print_text(text, width)
    puts ['|', text, '|'].pack("A2A#{width - 1}A1")
end

def print_empty_line(width)
    pretty_print_header_char('|', ' ', width)
end

def print_header_line(width)
    pretty_print_header_char('+', '-', width)
end
